def main(df,b):
    q1 = np.percentile(df[b],25)
    q2 = np.percentile(df[b],50)
    q3 = np.percentile(df[b],75)
    iqr = q3 - q1
    lower_fence = q1 - (iqr*1.5)
    upper_fence = q3 +  (iqr*1.5)
    lower_fence,upper_fence
    df[b] = np.where(((df[b]<lower_fence)|(df[b]>upper_fence)),q2,df[b])
